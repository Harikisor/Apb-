`define depth 64
`define size 16
`define addr_width $clog2(`depth)
module apbmaster(
  input valid,
  input [`addr_width-1:0]addr,
  input [`size-1:0]wdata,
  input reset,
  input clk,
  input wrtrd,
  input ready,
  input [`size-1:0]rdata,
  output reg valid_out,
  output reg [`addr_width-1:0]addr_out,
  output reg [`size-1:0]rout,
  output reg [`size-1:0]wdata_out,
  output reg wrtrd_out,
  output reg psel,
  output reg psel1,
  output reg penable,
  output reg [1:0] s,
  output reg [1:0] ns
);
  reg [1:0]state,next_state;
  reg  [`addr_width-1:0]addr_reg;
  localparam IDLE=2'b00,SETUP=2'b01,ACCESS=2'b10;
  always@(*)begin
    next_state = state;
    case(state)
      IDLE:begin
        if(valid)begin
          next_state=SETUP;
        end
        else begin
          next_state=IDLE;
        end
      end
      SETUP:
          next_state=ACCESS;
      ACCESS:begin
        if(!ready)begin
          next_state=ACCESS;
        end
      else if(ready && valid)begin
          next_state=SETUP;
      end
      else if(ready && !valid)begin
          next_state=IDLE;
      end
      end
    endcase
  end
  always@(posedge clk)begin
    if(reset)begin
      state<=IDLE;  
    end
    else
      state<=next_state;
  end
  always@(*)begin 
     valid_out=valid;
     wdata_out=wdata;
     psel=0;
     psel1=0; 
     penable=0;
     wrtrd_out=wrtrd;
    addr_out=addr_reg;
    case(state)
      IDLE:begin
        psel=0;
        psel1=0;
        penable=0;
      end
      SETUP:begin
        wdata_out=wdata;
        valid_out=valid;
        psel=(addr_reg<32);
        psel1=(addr_reg>31);
        penable=0;
      end
      ACCESS:begin
        psel  = (addr_reg<32);
        psel1 = (addr_reg>31);
        penable=1;
      end
    endcase
    s=state;
    ns=next_state;
  end
    always @(posedge clk) begin
    if(reset)
        rout <= 0;
    else if(state == ACCESS && !wrtrd && ready)
        rout <= rdata;
    end
  always @(posedge clk) begin
    if(reset)
        addr_reg <= 0;
    else if((state == IDLE && valid==1)||(state == SETUP && ready && valid))
        addr_reg <= addr;
end
endmodule
