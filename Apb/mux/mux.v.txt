module mux(
  input [`size-1:0]data1,
  input [`size-1:0]data,
  output reg [`size-1:0]data_out,
  input sel,
  input ready,
  input ready1,
  output reg ready_out
);
  always@(*)begin
    case(sel)
     0:begin
       ready_out=ready;
       data_out=data;
     end
      1:begin
         ready_out=ready1;
        data_out=data1;
      end
    endcase
  end
  endmodule
