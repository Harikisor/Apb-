class ahb_monitor extends uvm_monitor;
  `uvm_component_utils(ahb_monitor)
  virtual ahb_intf vif;
  uvm_analysis_port #(ahb_tx) item_collected_port;

  function new(string name, uvm_component parent);
    super.new(name, parent);
    item_collected_port = new("item_collected_port", this);
  endfunction

  function void build_phase(uvm_phase phase);
    if(!uvm_config_db#(virtual ahb_intf)::get(this, "", "vif", vif))
      `uvm_fatal("MON", "Could not get vif")
  endfunction

  task run_phase(uvm_phase phase);
    forever begin
      ahb_tx tx;
      @(posedge vif.clk);
      if (vif.ready && vif.valid) begin
        tx = ahb_tx::type_id::create("tx");
        tx.addr  = vif.o_haddr;
        tx.write = vif.o_hwrite;
        tx.wdata = vif.o_hwdata;
        tx.rdata = vif.o_hrdata;
        item_collected_port.write(tx);
      end
    end
  endtask
endclass
