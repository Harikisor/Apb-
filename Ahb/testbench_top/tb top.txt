module tb;
  logic clk, reset;
  ahb_intf vif(clk, reset);

  top dut (
    .clk(vif.clk), .reset(vif.reset), .write(vif.write), .burst(vif.burst),
    .addr(vif.addr), .wdata(vif.wdata), .hsize(vif.hsize), .start(vif.start)
  );

  // Sampling Master outputs via DUT hierarchy
  assign vif.o_hwdata = dut.a1.o_hwdata;
  assign vif.o_hrdata = dut.a1.o_hrdata;
  assign vif.o_hwrite = dut.a1.o_hwrite;
  assign vif.o_haddr  = dut.a1.o_haddr;
  assign vif.htrans   = dut.a1.htrans;
  assign vif.ready    = dut.ready_out;

  initial begin clk = 0; forever #5 clk = ~clk; end
  initial begin reset = 1; #20 reset = 0; end

  initial begin
    uvm_config_db#(virtual ahb_intf)::set(null, "*", "vif", vif);
    run_test("ahb_test");
  end

  initial begin $dumpfile("dump.vcd"); $dumpvars; end
endmodule
