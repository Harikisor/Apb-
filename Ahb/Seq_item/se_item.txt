class ahb_tx extends uvm_sequence_item;
  rand bit [9:0] addr;
  rand bit [31:0] wdata;
  rand bit write;
  bit [31:0] rdata; 
  rand bit [2:0] hsize;
  rand bit [2:0] burst;
  bit ready;
  bit [2:0]htrans;
  bit [9:0]add;
  constraint addr_align { addr[1:0] == 2'b00; } // Word Alignment
  constraint valid_hsize { hsize == 3'b010; }   // WORD size
  
  `uvm_object_utils(ahb_tx)
  function new(string name="ahb_tx"); super.new(name); endfunction
endclass
