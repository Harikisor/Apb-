class ahb_scoreboard extends uvm_scoreboard;
  `uvm_component_utils(ahb_scoreboard)
  
  uvm_analysis_imp #(ahb_tx, ahb_scoreboard) item_collected_export;
  bit [31:0] ref_mem [1024:0]; 
  bit [31:0] ref_mem1 [1024:0];
  function new(string name, uvm_component parent);
    super.new(name, parent);
    item_collected_export = new("item_collected_export", this);
  endfunction
  logic [31:0]expected ;
  function void write(ahb_tx tx);
    if(tx.addr<512) begin
         if (tx.write) begin
      ref_mem[tx.add] = tx.wdata;
      `uvm_info("SCB", $sformatf("STORE: Addr:%h Data:%h", tx.add, tx.wdata), UVM_LOW)
    end 
    else if(!tx.write)begin
      expected=ref_mem[tx.add];
      if (expected == tx.rdata) begin
          `uvm_info("SCB", $sformatf("PASS: Addr:%h Data:%h", tx.add, tx.rdata), UVM_LOW)
        end 
       else begin
         $display("FAIL:time:%t Addr:%h Data:%h expected:%h", $realtime,tx.add, tx.rdata ,expected);
        end
      end 
    end
    else if(tx.addr>511) begin
         if (tx.write) begin
      ref_mem1[tx.add] = tx.wdata;
      `uvm_info("SCB", $sformatf("STORE: Addr:%h Data:%h", tx.add, tx.wdata), UVM_LOW)
    end 
    else if(!tx.write)begin
      expected=ref_mem1[tx.add];
      if (expected == tx.rdata) begin
          `uvm_info("SCB", $sformatf("PASS: Addr:%h Data:%h", tx.add, tx.rdata), UVM_LOW)
        end 
       else begin
         $display("FAIL: time:%t Addr:%h Data:%h expected:%h",$realtime, tx.add, tx.rdata ,expected);
        end
      end 
    end
  endfunction
endclass
